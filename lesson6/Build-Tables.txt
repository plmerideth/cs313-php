mysql> CREATE DATABASE cs313
        CHARACTER SET utf8
        COLLATE utf8_unicode_ci;
        

mysql> use cs313;


mysql> SHOW TABLES;


CREATE TABLE accounts (
  acct_id TINYINT UNSIGNED NOT NULL AUTO_INCREMENT,
  first_name VARCHAR(20) NOT NULL,
  last_name VARCHAR(20) NOT NULL,
  email VARCHAR(60) NOT NULL,
  password CHAR(40) NOT NULL,
  datetime DATETIME NOT NULL,
  PRIMARY KEY (`acct_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

CREATE TABLE comments (
  comment_id TINYINT UNSIGNED NOT NULL AUTO_INCREMENT,
  comment TEXT NOT NULL,
  datetime DATETIME NOT NULL,
  pic_id TINYINT UNSIGNED NOT NULL,
  acct_id TINYINT UNSIGNED NOT NULL,
  PRIMARY KEY (comment_id),
  FOREIGN KEY (pic_id) REFERENCES pics (pic_id) ON DELETE NO ACTION ON UPDATE NO ACTION,
  FOREIGN KEY (acct_id) REFERENCES accounts (acct_id) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;


CREATE TABLE pics (
  pic_id TINYINT UNSIGNED NOT NULL AUTO_INCREMENT,
  filename VARCHAR(100) NOT NULL,
  datetime DATETIME NOT NULL,
  acct_id TINYINT UNSIGNED NOT NULL,
  cat_id TINYINT UNSIGNED NOT NULL,
  PRIMARY KEY (pic_id),
  FOREIGN KEY (acct_id) REFERENCES accounts (acct_id) ON DELETE NO ACTION ON UPDATE NO ACTION,
  FOREIGN KEY (cat_id) REFERENCES categories (cat_id) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;

CREATE TABLE categories (
  cat_id TINYINT UNSIGNED AUTO_INCREMENT,
  name VARCHAR(30),
  PRIMARY KEY(cat_id)
) ENGINE=INNODB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;
